class cv::AgastFeatureDetector

Wrapping class for feature detection using the AGAST method. :

#include <features2d.hpp>

class AgastFeatureDetector: public cv::Feature2D
{
public:
    // enums

    enum
    {
        AGAST_5_8          = 0,
        AGAST_7_12d        = 1,
        AGAST_7_12s        = 2,
        OAST_9_16          = 3,
        THRESHOLD          = 10000,
        NONMAX_SUPPRESSION = 10001,
    };

    // methods

    static
    Ptr<AgastFeatureDetector>
    create(
        int threshold = 10,
        bool nonmaxSuppression = true,
        int type = AgastFeatureDetector::OAST_9_16
        );

    virtual
    bool
    getNonmaxSuppression() const = 0;

    virtual
    int
    getThreshold() const = 0;

    virtual
    int
    getType() const = 0;

    virtual
    void
    setNonmaxSuppression(bool f) = 0;

    virtual
    void
    setThreshold(int threshold) = 0;

    virtual
    void
    setType(int type) = 0;
};
