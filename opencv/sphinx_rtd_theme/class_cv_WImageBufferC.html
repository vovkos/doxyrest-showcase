template class cv::WImageBufferC

Like a WImageBuffer class but when the number of channels is known at compile time.

#include <wimage.hpp>

template <
    typename T,
    int C
    >
class WImageBufferC: public cv::WImageC
{
public:
    // typedefs

    typedef WImage<T>::BaseType BaseType;

    // enums

    enum
    {
        kChannels = C,
    };

    // construction

    WImageBufferC();

    WImageBufferC(
        int width,
        int height
        );

    WImageBufferC(IplImage* img);

    // methods

    void
    Allocate(
        int width,
        int height
        );

    void
    CloneFrom(const WImageC<T, C>& src);

    bool
    IsNull() const;

    void
    ReleaseImage();

    void
    SetIpl(IplImage* img);
};
