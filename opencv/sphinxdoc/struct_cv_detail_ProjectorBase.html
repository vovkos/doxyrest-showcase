struct cv::detail::ProjectorBase

Base class for warping logic implementation.

struct ProjectorBase
{
    // fields

    float k[9];
    float k_rinv[9];
    float r_kinv[9];
    float rinv[9];
    float scale;
    float t[3];

    // methods

    void
    setCameraParams(
        InputArray K = Mat::eye(3, 3, CV_32F),
        InputArray R = Mat::eye(3, 3, CV_32F),
        InputArray T = Mat::zeros(3, 1, CV_32F)
