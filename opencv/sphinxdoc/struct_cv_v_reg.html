    typename _Tp,
    int n
    >
struct v_reg
{
    // construction

    v_reg(const _Tp* ptr);

    v_reg(
        _Tp s0,
        _Tp s1
        );

    v_reg(
        _Tp s0,
        _Tp s1,
        _Tp s2,
        _Tp s3
        );

        _Tp s0,
        _Tp s1,
        _Tp s2,
        _Tp s3,
        _Tp s4,

        _Tp s7
        );

        _Tp s1,
        _Tp s2,

};

Methods

_Tp
get0() const

Access first value.

Returns value of the first lane according to register type, for example:

v_int32x4 r(1, 2, 3, 4);
int v = r.get0(); // returns 1
v_uint64x2 r(1, 2);
uint64_t v = r.get0(); // returns 1
