class Poco::AbstractTimerCallback

This is the base class for all instantiations of the TimerCallback template.

#include <Timer.h>

class AbstractTimerCallback
{
public:
    // construction

    AbstractTimerCallback();
    AbstractTimerCallback(const AbstractTimerCallback& callback);

    // methods

    AbstractTimerCallback&
    operator=(const AbstractTimerCallback& callback);

    virtual
    void
    invoke(Timer& timer) const = 0;

    virtual
    AbstractTimerCallback*
    clone() const = 0;
};

// direct descendants

template <class C>
class TimerCallback;
