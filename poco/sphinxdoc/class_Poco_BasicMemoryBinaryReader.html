template class Poco::BasicMemoryBinaryReader

A convenient wrapper for using Buffer and MemoryStream with BinaryReader.

#include <BinaryReader.h>

template <typename T>
class BasicMemoryBinaryReader: public Poco::BinaryReader
{
public:
    // construction

    BasicMemoryBinaryReader(
        const Buffer<T>& data,
        StreamByteOrder byteOrder = NATIVE_BYTE_ORDER
        );

    BasicMemoryBinaryReader(
        const Buffer<T>& data,
        TextEncoding& encoding,
        StreamByteOrder byteOrder = NATIVE_BYTE_ORDER
        );

    // methods

    const Buffer<T>&
    data() const;

    const MemoryInputStream&
    stream() const;

    MemoryInputStream&
    stream();
};
