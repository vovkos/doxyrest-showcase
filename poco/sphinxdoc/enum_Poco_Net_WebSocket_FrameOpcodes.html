enum FrameOpcodes
{
    FRAME_OP_CONT    = 0x00,
    FRAME_OP_TEXT    = 0x01,
    FRAME_OP_BINARY  = 0x02,
    FRAME_OP_CLOSE   = 0x08,
    FRAME_OP_PING    = 0x09,
    FRAME_OP_PONG    = 0x0a,
    FRAME_OP_BITMASK = 0x0f,
    FRAME_OP_SETRAW  = 0x100,
};
