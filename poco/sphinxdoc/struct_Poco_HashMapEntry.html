This class template is used internally by HashMap.

    class Key,

{

    Value second;

    // construction

    HashMapEntry();

    HashMapEntry(
        const Key& key,
        const Value& value
        );

    // methods

    bool
    operator==(const HashMapEntry& entry) const;

    bool
    operator!=(const HashMapEntry& entry) const;
};
