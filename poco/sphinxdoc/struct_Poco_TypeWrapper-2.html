template struct Poco::TypeWrapper

Use the type wrapper if you want to decouple constness and references from template types.

#include <MetaProgramming.h>

template <typename T>
struct TypeWrapper
{
    // typedefs

    typedef T TYPE;
    typedef const T CONSTTYPE;
    typedef T& REFTYPE;
    typedef const T& CONSTREFTYPE;
};
